import httpx
from fastapi import FastAPI, Header, HTTPException
from typing import Optional, List

app = FastAPI()

@app.post("/initialize")
async def initialize_mcp():
    """Declares 2026 V28 & Cohort capabilities."""
    return {
        "jsonrpc": "2.0",
        "id": 1,
        "result": {
            "capabilities": {
                "extensions": {
                    "ai.promptopinion/fhir-context": {
                        "scopes": [
                            {"name": "patient/*.rs", "required": True},
                            {"name": "system/Patient.rs", "required": True}
                        ]
                    }
                }
            }
        }
    }

@app.post("/mcp/tools/audit_v28_cohort")
async def audit_v28_cohort(
    x_fhir_server_url: Optional[str] = Header(None, alias="X-FHIR-Server-URL"),
    x_fhir_access_token: Optional[str] = Header(None, alias="X-FHIR-Access-Token")
):
    """Fetches high-risk cohorts for V28 specificity auditing."""
    if not x_fhir_server_url:
        raise HTTPException(status_code=400, detail="Missing SHARP context.")

    headers = {"Accept": "application/fhir+json"}
    if x_fhir_access_token:
        headers["Authorization"] = f"Bearer {x_fhir_access_token}"

    try:
        async with httpx.AsyncClient() as client:
            # Fetch first 10 patients in the system
            cohort_resp = await client.get(f"{x_fhir_server_url}/Patient?_count=10", headers=headers)
            patients = cohort_resp.json().get("entry", [])
            
            payload = []
            for entry in patients:
                p_id = entry["resource"]["id"]
                # Fetch notes for each patient
                docs = await client.get(f"{x_fhir_server_url}/DocumentReference?subject=Patient/{p_id}", headers=headers)
                conds = await client.get(f"{x_fhir_server_url}/Condition?subject=Patient/{p_id}", headers=headers)
                
                payload.append({
                    "patient_id": p_id,
                    "notes_raw": docs.json(),
                    "current_conditions": conds.json()
                })
            return {"status": "success", "data": payload}
    except Exception as e:
        raise HTTPException(status_code=502, detail=f"FHIR Error: {str(e)}")

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=10000)
